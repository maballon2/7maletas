<script>

const state={nombre:'',edad:'',genero:'',poder:'',ubicacion:'',problemas:'',probTop:'',solucion:'',diferenciales:'',difUnico:'',testimonios:'',testmet:'',objeciones:'',objTop:'',garantia:'',gar:''};


function goTo(n){

  collect();

  document.querySelectorAll('.step').forEach(s=>s.classList.remove('active'));

  document.getElementById('s'+n).classList.add('active');

  const pct=Math.round((Math.max(n-1,0)/7)*100);

  document.getElementById('prog').style.width=pct+'%';

  window.scrollTo({top:0,behavior:'smooth'});

}


function collect(){

  const g=id=>document.getElementById(id);

  if(g('nombre'))state.nombre=g('nombre').value;

  if(g('ubicacion'))state.ubicacion=g('ubicacion').value;

  if(g('problemas'))state.problemas=g('problemas').value;

  if(g('prob-top'))state.probTop=g('prob-top').value;

  if(g('solucion'))state.solucion=g('solucion').value;

  if(g('diferenciales'))state.diferenciales=g('diferenciales').value;

  if(g('dif-unico'))state.difUnico=g('dif-unico').value;

  if(g('testimonios'))state.testimonios=g('testimonios').value;

  if(g('objeciones'))state.objeciones=g('objeciones').value;

  if(g('obj-top'))state.objTop=g('obj-top').value;

  if(g('garantia'))state.garantia=g('garantia').value;

}


function selectCard(el,key){

  el.parentElement.querySelectorAll('.card-opt').forEach(c=>c.classList.remove('selected'));

  el.classList.add('selected');

  state[key]=el.querySelector('.opt-label').textContent;

}


function finish(){

  collect();

  document.getElementById('nombre-out').textContent=(state.nombre||'cliente').split(' ')[0];

  const items=[

    {label:'Público',val:[state.edad,state.genero,state.poder].filter(Boolean).join(' · ')||'—'},

    {label:'Problema clave',val:state.probTop||state.problemas.split('\n')[0]||'—'},

    {label:'Solución',val:state.solucion?state.solucion.substring(0,60)+'…':'—'},

    {label:'Diferencial único',val:state.difUnico||'—'},

    {label:'Objeción principal',val:state.objTop||'—'},

    {label:'Garantía vs competencia',val:state.gar||'—'},

  ];

  document.getElementById('resumen').innerHTML=items.map(i=>`

    <div class="s-card">

      <div class="s-label">${i.label}</div>

      <div class="s-val">${i.val}</div>

    </div>`).join('');

  document.querySelectorAll('.step').forEach(s=>s.classList.remove('active'));

  document.getElementById('s8').classList.add('active');

  document.getElementById('prog').style.width='100%';

}

</script>
